## Solution
```
#include <string>
#include <stack>    // stack

using namespace std;

bool solution(string s){
    bool answer = true;
    
    stack<int> st;
    for(int i = 0; i < s.size(); i++){
        if(!st.empty() && (st.top() == '(') && s[i] == ')'){
            st.pop();
            continue;
        }
        
        st.push(s[i]);
    }

    answer = st.empty();
    return answer;
}
```
